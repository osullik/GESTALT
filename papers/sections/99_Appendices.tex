\subsection{Membership Search Algorithms}

\begin{algorithm}[H]
    \caption{Rank}\label{alg:rank}
    \begin{algorithmic}
        \State{\textit{\textbf{QT} a list of query terms to search for}}
        \State{\textit{\textbf{II} an inverted index with objects as keys and locations as values}}
        \State{- - - - -}
        \Procedure{Rank}{$S$,$QT$,$II$}\Comment{$S$ results from $Q$ query terms in $II$}
            \State{$LR$ = Empty Ordered Dictionary}\Comment{\textbf{L}ocation \textbf{R}ank}
            \For{Each $L$ in $S$}\Comment{\textbf{L}ocation}
                \State{$p$ = 1}%\Comment{Probability location correct}
                \For{$O$ in $QT$}
                    %\State{$OP$ = Prob of Obj@Locin $II$}\Comment{$OP$ = \textbf{O}bject \textbf{P}robability}
                    \State{$p$ = $p$x$OP$}\Comment{$OP$ = \textbf{O}bject \textbf{P}robability}
                \EndFor
                \State{$LR$[$L$] = $p$}
            \EndFor
            \State{\textbf{return} $LR$ ordered most to least likely}
        \EndProcedure
    \end{algorithmic}
\end{algorithm}

%\subsection{Concept Map Creation}
\begin{algorithm}[H]
    \caption{Creating a Concept Map}\label{alg:geoToGrid}
    \begin{algorithmic}
        \State{\textit{\textbf{T} a Location Table with ID, Name, Lat \& Long of its objects}}
        \State{- - - - -}
        \Procedure{createConceptMap}{$T$}
            \State{LonList, LatList = []} \Comment{Lists}
            %\State{LatList = []} \Comment{A list}
            \State{$M$ = [[][]]} \Comment{A matrix of 0s}
            %\State{Sort rows from North to South}
            \For{$R$ in $T$} \Comment{Where $T$ sorted North to South}     
                \State{Add R.$ID$ to $LatList$} \Comment{$R.ID$ is the obj unique ID}
            \EndFor
            %\State{Sort rows from West to East}         
            \For{$R$ in $T$}  \Comment{Where $T$ sorted West to East}  
                \State{Add R.$ID$ to $LonList$}
            \EndFor          
            \For{$ID$ in $Lonlist$}          
                \State{get index of $ID$ in $LonList$ as $i$}
                \State{get index of $ID$ in $LatList$ as $j$}               
                \State{$M$[$i$][$j$] = $ID.Name$} \Comment{Name of object with $ID$}
            \EndFor        
            \State{\textbf{return} $M$} \Comment{Having $[0,0]$ as NW corner}
        \EndProcedure
    \end{algorithmic}
\end{algorithm}


%\subsection{Object-Object Search Term Ordering}
\begin{algorithm}[H]
    \caption{Ordering the Search Terms}\label{alg:orderSearchTerms}
    \begin{algorithmic}
        \State{\textit{\textbf{M} A ConceptMap Matrix with objects or 0s}}
        \State{- - - - -}
        \Procedure{Order Search Terms}{$M$}
            \If{$M$ has single Item}
                \State{Return $M$}
            \EndIf
            \State{$Traversed$ = []}
            %\State{$R$, $C$ = Number of rows and columns}
            %\State{$Rd$ = ($R$+$C$)-1}
            \For{$r$ in range ($R$+$C$)-1} \Comment{$R$, $C$ = Number of rows and columns}
                \For{$i$ in range $r$+1}
                    %\State{$j$ = $r$-$i$}
                    \If{$i$ < $R$ and $r$-$i$ < $C$ and $M$[$i$][$r$-$i$] != 0}
                        %\If{$M$[$i$][$j$] != 0}
                            \State{Add $M$[$i$][$r$-$i$] to $Traversed$}
                        %\EndIf
                    \EndIf 
                \EndFor
            \EndFor
            \State{\textbf{return} $Traversed$}
        \EndProcedure
    \end{algorithmic}
\end{algorithm}

%\subsection{Location-Centric Strucutre Generation}

\begin{algorithm}[H]
    \caption{Generating Location Centric Structure}\label{alg:locationCentricGeneration}
    \begin{algorithmic}
        \State{\textit{\textbf{T} a Location Table with ID, Name, Lat \& Long of its objects}}
        \State{\textit{\textbf{X} a (x,y) tuple of the centroid for the location}}
        \State{- - - - -}
        \Procedure{MakeLocationCentricStructure}{$T$,$X$}
            \State{$Q$\{$NW$:[], $NE$:[], $SW$:[], $SE$:[]\}}\Comment{Dictionary}
            \For{$R$ in $T$}\Comment{R is a row describing a single object}
                \If{$R.y$ < $X.y$}
                    \If{$R.x$ <= $X.x$}
                        \State{Add $R.Name$ to $Q.SW$}
                    \Else
                        \State{Add $R.Name$ to $Q.SE$}
                    \EndIf
                \Else
                    \If{$R.x$ <= $X.x$} 
                        \State{Add $R.Name$ to $Q.NW$}
                    \Else
                        \State{Add $R.Name$ to $Q.NE$}
                    \EndIf
                \EndIf 
            \EndFor
        \EndProcedure
    \end{algorithmic}
\end{algorithm}
